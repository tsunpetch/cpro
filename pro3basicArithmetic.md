#basic Arithmetic

```
#include <stdio.h>
#include <stdlib.h>

int main(){

    printf("%d\n",1 + 2);
    printf("%d\n",10 % 2);    
    printf("%d\n",1 * (2+5) /6); 
    printf("%f\n",10 / 2.5); 
    printf("%d\n",10 / 2); 

    int num = 2;
    printf("%d\n",num);
    num += 5;
    printf("%d\n",num);
    num--;
    printf("%d\n",num);


    return 0;
}
```
