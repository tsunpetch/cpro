#Iteration

```
#include <stdio.h>
#include <stdlib.h>

int main(){
    
    //int i=6;
    for(int i=1;i<5;i++){
        printf("for round %d\n",i);
    }


    int j = 1;
    //int j = 6;
    while(j<5){
        printf("while round %d\n",j);
        j++;
    }

    int k = 1;
    //int k = 6;
    do{
        printf("do.. while round %d\n",k);
        k++;
    }while(k<5);





    return 0;
}
```
